===============
    v1.1.0
===============

-- Performance optimisations
   * HTML5/JS performance greatly improved (300%-400%)
   * YYC targets should also see performance gains


-- Changed standard for creating property setter scripts
   * tween's target id is now passed to script as argument1 (use of target id not required if property variable is global)
   * PLEASE UPDATE YOUR CUSTOM PROPERTY SETTERS!!!
   * e.g.
	// OLD
	x = argument0;

	// NEW
	argument1.x = argument0;


-- TweenCreate(), TweenPlay(), TweenDestroyWhenDone(), and TweenIsStopped() are now deprecated
   * Please remove deprecated versions, as they will be fully removed from TweenGMS Lite in a future update


-- TweenFire() script definition has been modified
   * Now takes a parameter for tween delays
   * Now takes a TWEEN_MODE_* macro for setting play mode
   * TweenFireBounce() and TweenFirePatrol() are now deprecated

	[Definition]
	TweenFire(target,property,ease,mode,delta,delay,dur,start,dest)

	e.g.
	TweenFire(id, x__, EaseOutQuad, TWEEN_MODE_PATROL, true, 0.0, 2.5, x, mouse_x);

	[Play Mode Macros]
	0 = TWEEN_MODE_ONCE
	1 = TWEEN_MODE_BOUNCE
	2 = TWEEN_MODE_PATROL

	[Available in TweenGMS Pro]
	3 = TWEEN_MODE_LOOP
	4 = TWEEN_MODE_REPEAT


-- Added new TweenAddCallback() script which use TWEEN_EV_**** macros:
   * TweenOnFinish and TweenOnBounce() scripts are now deprecated
   
	[Definition]
	TweenAddCallback(tween,event,target,script,arg0,arg1,...)

	e.g.
	TweenAddCallback(tween, TWEEN_EV_FINISH, id, ShowMessage, "Yeeeeehaw!");

	[Event Macros]
	TWEEN_EV_FINISH
	TWEEN_EV_CONTINUE  -- replaces "bounce" event

	[Available in TweenGMS Pro]
	TWEEN_EV_PLAY
	TWEEN_EV_STOP
	TWEEN_EV_PAUSE
	TWEEN_EV_RESUME
	TWEEN_EV_REVERSE
	
	TWEEN_EV_FINISH_DELAY
	TWEEN_EV_STOP_DELAY
	TWEEN_EV_PAUSE_DELAY
	TWEEN_EV_RESUME_DELAY


-- Added basic example objects

-- Improved error reporting when using invalid tween id

-- Adjusted TweenSetEase() to force property update
